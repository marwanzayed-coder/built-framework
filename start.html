Functions

Push Function

import { push } from "built-javascript";

/*
  ==============
  ==== Push ====
  ==============
  push(array, ele);
*/

let array = [1, 2, 3];
push(array, 4, 5, 6); // [1, 2, 3, 4, 5, 6]
Join Function

import { join } from "built-javascript";

/*
  ==============
  ==== Join ====
  ==============
  join(array, splitter);
*/

let array = [1, 2, 3];
join(array, " "); // 1 2 3

join(array); // 123
Length Function

import { length } from "built-javascript";

/*
  ================
  ==== Length ====
  ================
  length(element);
    - The element can be string or array
*/

let array = [1, 2, 3];
length(array); // 3

let name = "marwan"
length(name); // 6
Reverse Function

import { reverse } from "built-javascript";

/*
  =================
  ==== Reverse ====
  =================
  reverse(string);
*/

let name = "marwan"
reverse(name); // nawram
